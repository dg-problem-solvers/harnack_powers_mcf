\begin{abstract}
%We prove differential Harnack inequalities for weakly convex hypersurfaces on the unit sphere shrinking by positive powers of homogeneous degree one, inverse-concave functions of principal curvatures, $f^{\alpha}>0,~\alpha\geq 1$ . As an application, we prove that the only convex, ancient solutions of curvature flows with $f$ concave and inverse-concave velocities are shrinking geodesic spheres.

\maketitle

\section{Introduction}

We consider the evolution of a hypersurface $M^n$ by
\[\partial_tx=-\varphi\nu,~ x:M^n\times[0,T)\to M_K,\]
where \(M_K\) is the simply connected space form of constant sectional curvature \(K\) and where $\varphi$ is a function on positive symmetric matrices, evaluated at the Weingarten map,
\[
\varphi = \varphi(\mathcal{W}) = \varphi(h^j_i).
\]
We sometimes think of \(\varphi\) as a function of the metric and second fundamental form
\[
\varphi(g, h) = \varphi(g^{jk} h_{ik}).
\]

We are particularly interested in the case \(\varphi = f^{\alpha}\) where \(f\) is homogeneous degree one, inverse concave (concave also?) and \(\alpha \in (0,1)\). 

Our principal result is a Harnack inequality for these flows, extending the inequality in \cite{2015arXiv150802821B, bryanlouie} and inspired by \cite{MR1316556, MR1100812, MR1296393, MR1480081}. Namely, we prove that if \(\varphi = H^{\alpha}\) for \(0 < \alpha \leq 1\) we have

\begin{theorem}
If $\frac{1}{2}+\frac{1}{2n}\leq {\alpha}< 1,$ then
\[
\partial_tH^{\alpha} - b^{ij}\nabla_iH^{\alpha}\nabla_jH^{\alpha} - \frac{K {\alpha}}{2{\alpha}-1}H^{2{\alpha}-1} + \frac{{\alpha}}{{\alpha}+1} \frac{H^{\alpha}}{t} > 0.
\]
If $0<{\alpha}\leq \frac{1}{2} + \frac{1}{2n}$ or ${\alpha}=1$, then
\[
\partial_tH^{\alpha} - b^{ij}\nabla_iH^{\alpha}\nabla_jH^{\alpha} - K n{\alpha}H^{2{\alpha}-1} + \frac{{\alpha}}{{\alpha}+1} \frac{H^{\alpha}}{t} > 0.
\]
\end{theorem}

\section{Evolution equations}
Define \(M_t:=x(M^n,t)\) and let $\nabla$ denote the Levi-Civita connection of $M_t$. Let 
\[
\varphi^{ij} = \fr{\partial\varphi}{\partial h_{ij}}, \quad \varphi^{ij,kl} = \fr{\partial^2\varphi}{\partial h_{kl} \partial h_{ij}}.
\]
and define the operator
\[
\Box = \varphi^{ij} \nabla_i \nabla_j
\]

The evolution equations in the following lemma are standard and can be found in many places. The basic tools are commuting derivatives using that \(\bar{R}_{ijkl} = K(\bar{g}_{ik}\bar{g}_{jl} - \bar{g}_{il}\bar{g}_{jk})\), the Gauss equation \(R_{ijkl} = \bar{R}_{ijkl} + h_{ik}h_{jl} - h_{jl}h_{ik}\) and the Codazzi equation \(\nabla_i h_{jk} = \nabla_j h_{ik}\). Compare \cite[p.~94-95]{Gerhardt:/2006} and the formula \cite[eq.~(6.17)]{Gerhardt:01/1996}.

\begin{lemma}
The following evolution equations and commutator identities hold
\begin{enumerate}
  \item $\partial_tg_{ij}=-2\varphi h_{ij}$
  \item $\partial_tg^{ij}=2\varphi g^{im}g^{jn}h_{ij}=2\varphi h^{ij}$
    \item $\partial_t h_{ij}=\nabla_j\nabla_i\varphi-\varphi(h^2)_{ij}+\lambda \varphi g_{ij}$
  \item $\partial_t h_i^j=\nabla^j\nabla_i\varphi+\varphi(h^2)_i^j+\lambda \varphi\delta_i^j$
  \item \begin{align*}
\partial_t h_{ij}=&\Box h_{ij}+\varphi^{kl}h_{rk}h_l^rh_{ij}-\varphi'fh_{ri}h_j^r-\varphi h_i^kh_{kj}\\
&+\varphi^{kl,rs}\nabla_ih_{kl}\nabla_jh_{rs}\\
&+\lambda \{(\varphi+\varphi'f)g_{ij}-\varphi^{kl}g_{kl}h_{ij}\}
\end{align*}
  \item \begin{align*}
\partial_t h_i^j=&\Box h_i^j+\varphi^{kl}h_{rk}h_l^rh_i^j-\varphi'fh_{ri}h^{rj}+\varphi h_i^kh_k^j\\
&+\varphi^{kl,rs}\nabla_ih_{kl}\nabla^jh_{rs}\\
&+\lambda \{(\varphi+\varphi'f)\delta_i^j-\varphi^{kl}g_{kl}h_i^j\}
\end{align*}
  \item \begin{align*}
\partial_t b^{ij} &= \Box b^{ij} - \p^{rs}h_{rk}h^{k}_{s}b^{ij}+\br{\p'f+\p}g^{ij}-\l\br{\p'f+\p}b^{ir}b^{j}_{r}+\l\p^{kl}g_{kl}b^{ij} \\
& \quad -\br{2b^{lq}\p^{kp}+\p^{kl,pq}}b^{ir}b^{js}h_{kl;r}h_{pq;s}
\end{align*}
where $(b^{ij})$ denotes the inverse of $(h_{ij}).$

  \item $\partial_t \varphi=\Box \varphi+\varphi\varphi^{ij}(h^2)_{ij}+\lambda \varphi\varphi^{ij}g_{ij}$
\end{enumerate}
\end{lemma}

\Theo{lemma}{EvGamma}{
The Christoffel symbols evolve according to
\eq{\dot{\G}^{k}_{ij}=-\p g^{kl}h_{ij;l}-g^{kl}\p_{;j}h_{li}-g^{kl}\p_{;i}h_{lj}+g^{kl}\p_{;l}h_{ij}.}
}

\pf{
We have
\eq{\G^{k}_{ij}=\fr{1}{2}g^{kl}\br{g_{il,j}+g_{jl,i}-g_{ij,l}},}
where $_{,i}$ denotes partial differentiation with respect to $\xi^{i}$ in a coordinate system $(\xi^{i}).$
Since $\dot{\G}^{k}_{ij}$ is a tensor, we may a priorily consider normal coordinates in a given point. Then, using the Codazzi equations,
\eq{\dot{\G}^{k}_{ij}&=\fr{1}{2}g^{kl}\br{\dot{g}_{il,j}+\dot{g}_{jl,i}-\dot{g}_{ij,l}}\\
				&=-\p g^{kl}h_{ij;l}-g^{kl}\p_{;j}h_{il}-g^{kl}\p_{;i}h_{lj}+g^{kl}\p_{;l}h_{ij}.}
}



\Theo{lemma}{deltBox}{
There holds
\eq{\br{\del_{t}\Box-\Box\del_{t}}\p&=2\p^{ij}h^{k}_{i}\br{\p_{;kj}\p+\p_{;k}\p_{;j}}+\br{\p-\p^{ij}h_{ij}}|\nabla\p|^{2}\\
					&\hp{=}+\p^{ij,rs}\p_{;ij}\br{\p_{;rs}+\p h_{rm}h^{m}_{s}+\l\p g_{rs}}.}
}


\pf{Considering the curvature function $\p$ to depend on $(h_{ij})$ and $(g_{ij}),$
\eq{\p=\p(h_{ij},g_{ij}),}
we denote
\eq{\p^{ij}=\fr{\del \p}{\del h_{ij}},\quad \p^{ij,kl}=\fr{\del^{2}\p}{\del h_{ij}\del h_{kl}},}
which are tensors, in particular we have
\eq{\p^{i}_{j}\equiv \fr{\del \p}{\del h^{j}_{i}}=\p^{ik}g_{kj}.}
Furthermore we have
\eq{\fr{\del\p}{\del g_{ij}}=\p^{l}_{k}\fr{\del h^{k}_{l}}{\del g_{ij}}=-\p^{l}_{k}g^{ki}g^{rj}h_{rl}=-\p^{li}h^{j}_{l}.}
Thus
\eq{\label{deltBox1}\del_{t}\p^{ij}&=\p^{ij,kl}\dot{h}_{kl}+\fr{\del\p^{ij}}{\del g_{kl}}\dot{g}_{kl}\\
			&=\p^{ij,kl}\br{\dot{h}_{kl}+2\p h_{lm}h^{m}_{k}}+2\p\p^{jk}g^{li}h_{kl}\\
			&=\p^{ij,kl}\br{\dot{h}_{kl}+2\p h_{lm}h^{m}_{k}}+2\p\p^{jk}h^{i}_{k}.}
We obtain from \cref{EvGamma},
\eq{\br{\del_{t}\Box-\Box\del_{t}}\p&=\br{\del_{t}\p^{ij}}\p_{;ij}+\p^{ij}\br{\del_{t}\p_{;ij}-\dot{\p}_{;ij}}\\
							&=\br{\del_{t}\p^{ij}}\p_{;ij}-\p^{ij}\dot{\G}^{k}_{ij}\p_{;k}\\
							&=2\p h^{i}_{k}\p^{kj}\p_{;ij}+\p^{ij,rs}\br{\dot{h}_{rs}+2\p h_{rm}h^{m}_{s}}\p_{;ij}\\
							&\hp{=}+\p\p^{ij}h_{ij;l}g^{kl}\p_{;k}+2\p^{ij}g^{kl}h_{li}\p_{;j}\p_{;k}-\p^{ij}h_{ij}g^{kl}\p_{;l}\p_{;k}. }
	}	

\Theo{lemma}{Evbeta}{
The quantity 
\eq{\b=\dot{\p}-\l\p^{ij}g_{ij}\p=\Box\p+\p^{ij}h_{ik}h^{k}_{j}\p.}
satisfies
\eq{\br{\del_{t}-\Box}\b&=\p^{ij}h_{ik}h^{k}_{j}\b+2\p^{ij}h^{k}_{i}\br{2\p_{;kj}\p+\p_{;k}\p_{;j}+h^{s}_{j}h_{sk}\p^{2}}\\
				&\hp{=}+\br{\p-\p^{ij}h_{ij}}|\nabla\p|^{2}\\
				&\hp{=}+\p^{ij,kl}\br{\p_{;ij}+h_{ip}h^{p}_{j}\p}\br{\p_{;kl}+h^{m}_{k}h_{ml}\p}\\
				&\hp{=}+\l\p^{ij}g_{ij}\b+2\l\p^{rs}\br{\p^{ij}g_{ij}}_{;r}\p_{;s}+\l\Box\br{\p^{ij}g_{ij}}\p\\
				&\hp{=}+\l\p^{ij,rs}g_{rs}\br{\p_{;ij}\p+h_{ik}h^{k}_{j}\p^{2}}+2\l\p^{ij}h_{ij}\p^{2}.}
}

\pf{We will need the following auxillary calculation, where we use \cref{deltBox1}.
\eq{\del_{t}\br{\p^{ij}h_{ik}h^{k}_{j}}&=\del_{t}\p^{ij}h_{ik}h^{k}_{j}+\p^{ij}\dot{h}_{ik}h^{k}_{j}+\p^{ij}h_{ik}\dot{h}^{k}_{j}\\
				&=\p^{ij,kl}\br{\dot{h}_{kl}+2\p h_{kr}h^{r}_{l}}h_{ir}h^{r}_{j}+2\p\p^{jk}h^{i}_{k}h_{ir}h^{r}_{j}\\
				&\hp{=}+\p^{ij}\br{\p_{;ik}-\p h^{r}_{i}h_{rk}+\l\p g_{ik}}h^{k}_{j}\\
				&\hp{=}+\p^{ij}\br{{\p_{;j}}^{k}+\p h^{k}_{r}h^{r}_{j}+\l\p\d^{k}_{j}}h_{ik}\\
				&=\p^{ij,kl}\br{\p_{;kl}+\p h^{r}_{k}h_{rl}+\l\p g_{kl}}h_{is}h^{s}_{j}+2\p\p^{jk}h^{i}_{k}h_{ir}h^{r}_{j}\\
				&\hp{=}+2\p^{ij}h^{k}_{i}\p_{;jk}+2\l\p^{ij}h_{ij}\p.}

 We use \cref{deltBox} to compute
\eq{\br{\del_{t}-\Box}\b&=\del_{t}\Box\p+\del_{t}\br{\p^{ij}h_{ik}h^{k}_{j}\p}-\Box\Box\p-\Box\br{\p^{ij}h_{ik}h^{k}_{j}\p}\\
			&=\Box\br{\l\p^{ij}g_{ij}\p}+2\p^{ij}h^{k}_{i}\br{\p_{;kj}\p+\p_{;k}\p_{;j}}+\br{\p-\p^{ij}h_{ij}}|\nabla\p|^{2}\\
					&\hp{=}+\p^{ij,rs}\p_{;ij}\br{\p_{;rs}+\p h_{rm}h^{m}_{s}+\l\p g_{rs}}+\del_{t}\br{\p^{ij}h_{ik}h^{k}_{j}\p}\\
			&=\l\p^{ij}g_{ij}\b-\l\p^{kl}g_{kl}\p^{ij}h_{ir}h^{r}_{j}\p+2\l\p^{rs}\br{\p^{ij}g_{ij}}_{;r}\p_{;s}\\
			&\hp{=}+\l\Box\br{\p^{ij}g_{ij}}\p+2\p^{ij}h^{k}_{i}\br{\p_{;kj}\p+\p_{;k}\p_{;j}}\\
			&\hp{=}+\br{\p-\p^{ij}h_{ij}}|\nabla\p|^{2}+\p^{ij,rs}\p_{;ij}\br{\p_{;rs}+\p h_{rm}h^{m}_{s}+\l\p g_{rs}}\\
			&\hp{=}+\p^{ij}h_{ik}h^{k}_{j}\b+\l\p^{ij}h_{ik}h^{k}_{j}\p^{rs}g_{rs}\p+\del_{t}\br{\p^{ij}h_{ik}h^{k}_{j}}\p\\
			&=\p^{ij}h_{ik}h^{k}_{j}\b+\l\p^{ij}g_{ij}\b+2\l\p^{rs}\br{\p^{ij}g_{ij}}_{;r}\p_{;s}\\
			&\hp{=}+\l\Box\br{\p^{ij}g_{ij}}\p+2\p^{ij}h^{k}_{i}\br{\p_{;kj}\p+\p_{;k}\p_{;j}}	\\
			&\hp{=}+\br{\p-\p^{ij}h_{ij}}|\nabla\p|^{2}+\p^{ij,rs}\p_{;ij}\br{\p_{;rs}+\p h_{rm}h^{m}_{s}+\l\p g_{rs}}\\
			&\hp{=}+\p^{ij,kl}\br{\p_{;kl}+\p h^{r}_{k}h_{rl}+\l\p g_{kl}}h_{is}h^{s}_{j}\p+2\p^{jk}h^{i}_{k}h_{ir}h^{r}_{j}\p^{2}\\
				&\hp{=}+2\p^{ij}h^{k}_{i}\p_{;jk}\p+2\l\p^{ij}h_{ij}\p^{2}.}
}

\Theo{lemma}{gradBox}{
For every smooth function $F$  there holds
\eq{\br{\Box F}_{;i}-\Box F_{;i}&=\p^{kl,rs}h_{rs;i}F_{;kl}+\l\p^{kl}g_{ki}F_{;l}-\l\p^{kl}g_{kl}F_{;i}\\
					&\hp{=}+\p^{kl}h^{m}_{l}h_{ki}F_{;m}-\p^{kl}h_{kl}h^{m}_{i}F_{;m}.}
}

\pf{
From the Ricci identities we obtain
\eq{\br{\p^{kl}F_{;kl}}_{;i}-\p^{kl}F_{;ikl}=\p^{kl,rs}h_{rs;i}F_{;kl}+\p^{kl}{R^{m}}_{kli}F_{;m}}
and from the Gauss equation we obtain
\eq{\p^{kl}{R^{m}}_{kli}F_{;m}=\l\p^{kl}g_{ki}F_{;l}-\l\p^{kl}g_{kl}F_{;i}+\p^{kl}h^{m}_{l}h_{ki}F_{;m}-\p^{kl}h_{kl}h^{m}_{i}F_{;m}.}
}

\Theo{lemma}{Evgradphi}{
There holds
\eq{\br{\del_{t}-\Box}\p_{;i}&=\br{\p^{kl}h_{kr}h^{r}_{l}\p}_{;i}+\l\p^{kl}\p_{;l}g_{ki}+\l\p^{kl,rs}h_{rs;i}g_{kl}\p\\
				&\hp{=}+\p^{kl,rs}h_{rs;i}\p_{;kl}+\br{\p^{kl}h^{m}_{l}h_{ki}-\p^{kl}h_{kl}h^{m}_{i}}\p_{;m}.}
}

\pf{
\eq{\del_{t}\p_{;i}-\Box\p_{;i}&=\br{\Box\p}_{;i}+\br{\p^{kl}h_{rk}h^{r}_{l}\p}_{;i}+\l\br{\p^{kl}g_{kl}\p}_{;i}\\
					&\hp{=}-\br{\Box\p}_{;i}+\p^{kl,rs}h_{rs;i}\p_{;kl}+\l\p^{kl}g_{ki}\p_{;l}-\l\p^{kl}g_{kl}\p_{;i}\\
					&\hp{=}+\p^{kl}h^{m}_{l}h_{ki}\p_{;m}-\p^{kl}h_{kl}h^{m}_{i}\p_{;m}\\
					&=\br{\p^{kl}h_{rk}h^{r}_{l}\p}_{;i}+\l\p^{kl,rs}h_{rs;i}g_{kl}\p+\p^{kl,rs}h_{rs;i}\p_{;kl}\\
					&\hp{=}+\l\p^{kl}g_{ki}\p_{;l}+\br{\p^{kl}h^{m}_{l}h_{ki}-\p^{kl}h_{kl}h^{m}_{i}}\p_{;m}.}
}

\Theo{lemma}{Evtheta}{
The quantity $\t=b^{ij}\p_{;i}\p_{;j}$ satisfies
\eq{\br{\del_{t}-\Box}\t&=\p^{rs}h_{rk}h^{k}_{s}\t-\br{2b^{lq}\p^{kp}+\p^{kl,pq}}b^{ir}b^{js}\p_{;i}\p_{j}h_{kl;r}h_{pq;s}\\
							&\hp{=}+\br{\p'f+\p-2\p^{kl}h_{kl}}|\nabla\p|^{2}+2\p^{jl}h^{m}_{l}\p_{;m}\p_{;j}\\
							&\hp{=}+2b^{ij}\p^{kl,rs}h_{rs;i}\p_{;j}\br{\p_{;kl}+h_{rk}h^{r}_{l}\p}+4b^{ij}\p^{kl}h_{rk;i}h^{r}_{l}\p_{;j}\p\\
							&\hp{=}+4\p^{rs}b^{ip}b^{jq}h_{pq;r}\p_{;is}\p_{;j}-2b^{ij}\p^{rs}\p_{;ir}\p_{;js}\\
							&\hp{=}-\l\br{\p'f+\p}b^{ir}b^{j}_{r}\p_{;i}\p_{;j}+\l\p^{kl}g_{kl}\t+2\l b^{j}_{k}\p^{kl}\p_{;l}\p_{;j}\\
							&\hp{=}+2\l b^{ij}\p^{kl,rs}h_{rs;i}\p_{;j}g_{kl}\p.}
}

\pf{
\eq{\br{\del_{t}-\Box}\t&=\br{\dot{b}^{ij}-\Box b^{ij}}\p_{;i}\p_{;j}+2b^{ij}\br{\dot{\p}_{;i}-\Box\p_{;i}}\p_{;j}\\
							&\hp{=}-4\p^{rs}{b^{ij}}_{;r}\p_{;is}\p_{;j}-2b^{ij}\p^{rs}\p_{;ir}\p_{;js}\\
							&=-\p^{rs}h_{rk}h^{k}_{s}\t+\br{\p'f+\p}|\nabla\p|^{2}-\l\br{\p'f+\p}b^{ir}b^{j}_{r}\p_{;i}\p_{;j}\\
							&\hp{=}+\l\p^{kl}g_{kl}\t-\br{2b^{lq}\p^{kp}+\p^{kl,pq}}b^{ir}b^{js}\p_{;i}\p_{;j}h_{kl;r}h_{pq;s}\\
							&\hp{=}+2\p^{kl}h_{rk}h^{r}_{l}\t+2b^{ij}\br{\p^{kl}h_{rk}h^{r}_{l}}_{;i}\p_{;j}\p+2b^{ij}\p^{kl,rs}h_{rs;i}\p_{;j}\p_{;kl}\\
							&\hp{=}+2b^{ij}\br{\p^{kl}h^{m}_{l}h_{ki}-\p^{kl}h_{kl}h^{m}_{i}}\p_{;m}\p_{;j}\\
							&\hp{=}+2\l b^{ij}\p^{kl,rs}h_{rs;i}\p_{;j}g_{kl}\p+2\l b^{j}_{k}\p^{kl}\p_{;l}\p_{;j}\\
							&\hp{=}-4\p^{rs}{b^{ij}}_{;r}\p_{;is}\p_{;j}-2b^{ij}\p^{rs}\p_{;ir}\p_{;js}\\
							&=\p^{rs}h_{rk}h^{k}_{s}\t-\br{2b^{lq}\p^{kp}+\p^{kl,pq}}b^{ir}b^{js}\p_{;i}\p_{;j}h_{kl;r}h_{pq;s}\\
							&\hp{=}+\br{\p'f+\p}|\nabla\p|^{2}+2\p^{jl}h^{m}_{l}\p_{;m}\p_{;j}-2\p^{kl}h_{kl}|\nabla\p|^{2}\\
							&\hp{=}+2b^{ij}\p^{kl,rs}h_{rs;i}\p_{;j}\br{\p_{;kl}+h_{rk}h^{r}_{l}\p}+4b^{ij}\p^{kl}h_{rk;i}h^{r}_{l}\p_{;j}\p\\
							&\hp{=}-4\p^{rs}{b^{ij}}_{;r}\p_{;is}\p_{;j}-2b^{ij}\p^{rs}\p_{;ir}\p_{;js}\\
							&\hp{=}-\l\br{\p'f+\p}b^{ir}b^{j}_{r}\p_{;i}\p_{;j}+\l\p^{kl}g_{kl}\t+2\l b^{j}_{k}\p^{kl}\p_{;l}\p_{;j}\\
							&\hp{=}+2\l b^{ij}\p^{kl,rs}h_{rs;i}\p_{;j}g_{kl}\p.}
}

\Theo{lemma}{Evchi}{
Let $\d\neq 0.$ The quantity 
\eq{\chi=t(\b-\t)+\d\p}
satisfies
\eq{\dot{\chi}-\Box\chi&=\br{\fr{\b-\t}{\d\p}+\p^{ij}h_{ik}h^{k}_{j}+\l\p^{ij}g_{ij}}\chi\\
				&\hp{=}+t\br{\p^{ij,kl}+2b^{il}\p^{jk}-\fr{\p^{ij}\p^{kl}}{\d\p}}\eta_{ij}\eta_{kl}+t\l R,}
where
\eq{\eta_{ij}=\p_{;ij}-b^{rs}h_{ij;r}\p_{;s}+h_{ip}h^{p}_{j}\p}
and 
\eq{R&=2\p^{rs}\br{\p^{ij}g_{ij}}_{;r}\p_{;s}+\Box\br{\p^{ij}g_{ij}}\p\\
				&\hp{=}+\p^{kl,rs}g_{rs}\br{\p_{;kl}\p+h_{kr}h^{r}_{l}\p^{2}-2b^{ij}h_{kl;i}\p_{;j}\p}+2\p^{ij}h_{ij}\p^{2}\\
				&\hp{=}+\br{\p'f+\p}b^{ir}b^{j}_{r}\p_{;i}\p_{;j}-2 b^{j}_{k}\p^{kl}\p_{;l}\p_{;j}.
}
}
	
\pf{
\eq{\dot\chi&-\Box\chi=\b-\t+t\br{\br{\del_{t}-\Box}\br{\b-\t}}+\d\br{\dot\p-\Box\p}\\
				&=\b-\t+\d\p^{ij}h_{ik}h^{k}_{j}\p+\d\l\p^{ij}g_{ij}\p\\
				&\hp{=}+t\Big(\p^{ij}h_{ik}h^{k}_{j}\br{\b-\t}+\br{\p^{ij}h_{ij}-\p'f}|\nabla\p|^{2}\\
				&\hp{=}+\p^{ij,kl}\br{\p_{;ij}+h_{ip}h^{p}_{j}\p}\br{\p_{;kl}+h^{m}_{k}h_{ml}\p}\\
				&\hp{=}+\p^{ij,kl}b^{pr}b^{sq}\p_{;p}\p_{;q}h_{ij;r}h_{kl;s}-2\p^{ij,kl}b^{rs}h_{kl;r}\p_{;s}\br{\p_{;ij}+h_{ip}h^{p}_{j}\p}\\
				&\hp{=}+2\p^{ik}h^{r}_{i}\br{2\p_{;rk}\p+h^{s}_{k}h_{sr}\p^{2}}+2b^{il}\p^{jk}b^{pr}b^{qs}\p_{;p}\p_{;q}h_{kl;r}h_{ij;s}\\
				&\hp{=}-4b^{il}\p^{jk}h_{rk;i}h^{r}_{j}\p_{;l}\p-4\p^{jk}b^{il}b^{pq}h_{lj;q}\p_{;ik}\p_{;p}+2b^{il}\p^{jk}\p_{;ij}\p_{;lk}\\
				&\hp{=}+\l\p^{ij}g_{ij}\br{\b-\t}+2\l\p^{rs}\br{\p^{ij}g_{ij}}_{;r}\p_{;s}+\l\Box\br{\p^{ij}g_{ij}}\p\\
				&\hp{=}+\l\p^{kl,rs}g_{rs}\br{\p_{;kl}\p+h_{kr}h^{r}_{l}\p^{2}-2b^{ij}h_{kl;i}\p_{;j}\p}+2\l\p^{ij}h_{ij}\p^{2}\\
				&\hp{=}+\l\br{\p'f+\p}b^{ir}b^{j}_{r}\p_{;i}\p_{;j}-2\l b^{j}_{k}\p^{kl}\p_{;l}\p_{;j}\Big)\\
				&=\b-\t+\br{\p^{ij}h_{ik}h^{k}_{j}+\l\p^{ij}g_{ij}}\chi+t\br{\p^{ij}h_{ij}-\p'f}|\nabla\p|^{2}\\
				&\hp{=}+t\p^{ij,kl}\br{\p_{;ij}-b^{rs}h_{ij;r}\p_{;s}+h_{ip}h^{p}_{j}\p}\br{\p_{;kl}-b^{mn}h_{kl;m}\p_{;n}+h_{kq}h^{q}_{l}\p}\\
				&\hp{=}+2tb^{il}\p^{jk}\br{\p_{;ij}-b^{rs}h_{ij;r}\p_{;s}+h_{ip}h^{p}_{j}\p}\br{\p_{;kl}-b^{mn}h_{kl;m}\p_{;n}+h_{kq}h^{q}_{l}\p}\\
				&\hp{=}+2t\l\p^{rs}\br{\p^{ij}g_{ij}}_{;r}\p_{;s}+t\l\Box\br{\p^{ij}g_{ij}}\p\\
				&\hp{=}+t\l\p^{kl,rs}g_{rs}\br{\p_{;kl}\p+h_{kr}h^{r}_{l}\p^{2}-2b^{ij}h_{kl;i}\p_{;j}\p}+2t\l\p^{ij}h_{ij}\p^{2}\\
				&\hp{=}+t\l\br{\p'f+\p}b^{ir}b^{j}_{r}\p_{;i}\p_{;j}-2t\l b^{j}_{k}\p^{kl}\p_{;l}\p_{;j}.
}
}
